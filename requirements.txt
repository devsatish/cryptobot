# Exception Groups and the new `except*` syntax
try:
    # Some code that may raise multiple exceptions
    pass
except* (ValueError, TypeError) as e:
    print(f"Caught an exception group: {e}")

# `Self` type in class methods for better type hinting

        self.value = value
        return self

# Variadic Generics for more flexible type annotations
from typing import TypeVar, Generic

T = TypeVar('T')

class Container(Generic[T]):
    def __init__(self, *items: T):
        self.items = items

    def add_item(self, item: T) -> None:
        self.items += (item,)
